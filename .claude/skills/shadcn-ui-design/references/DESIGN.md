# Design System Reference — shadcn/ui + Tailwind CSS v4 + Next.js

## Stack

| Layer | Tech | Notes |
|---|---|---|
| Framework | Next.js 15 (App Router) | Server Components by default |
| Styling | Tailwind CSS v4 | CSS-first config |
| Components | shadcn/ui (new-york style) | Code lives in `components/ui/` |
| Language | TypeScript strict | No `any`, explicit interfaces |
| Dark Mode | next-themes | 3 modes: light / dark / primary |
| Forms | react-hook-form + zod | Always use together |
| Icons | lucide-react | Only icon library |

---

## Rules — Always Follow

```
DO                                      DON'T
──────────────────────────────────────────────────────────────────
bg-background / text-foreground         bg-white / text-gray-900
bg-primary / text-primary-foreground    bg-neutral-900 / bg-violet-600
text-muted-foreground                   text-gray-500 / text-neutral-400
bg-destructive                          text-red-500 / bg-red-600
border-border                           border-gray-200
font-sans (Inter)                       font-["Inter"]
font-mono (Geist Mono)                  font-["JetBrains_Mono"]
cn("base", cond && "extra")             `base ${cond ? "extra" : ""}`
interface ComponentProps { ... }        props: any / no types
"use client" only when needed           "use client" on every file
components/ui/ — read only              edit shadcn generated files
```

**Never use violet/purple colors** — primary palette is neutral (neutral/900 light, neutral/200 dark).

### Theme Toggle

```tsx
"use client"
import { Moon, Sun } from "lucide-react"
import { useTheme } from "next-themes"
import { Button } from "@/components/ui/button"
import { DropdownMenu, DropdownMenuContent, DropdownMenuItem, DropdownMenuTrigger } from "@/components/ui/dropdown-menu"

export function ModeToggle() {
  const { setTheme } = useTheme()
  return (
    <DropdownMenu>
      <DropdownMenuTrigger asChild>
        <Button variant="outline" size="icon">
          <Sun className="h-4 w-4 rotate-0 scale-100 dark:-rotate-90 dark:scale-0 transition-all" />
          <Moon className="absolute h-4 w-4 rotate-90 scale-0 dark:rotate-0 dark:scale-100 transition-all" />
          <span className="sr-only">Toggle theme</span>
        </Button>
      </DropdownMenuTrigger>
      <DropdownMenuContent align="end">
        <DropdownMenuItem onClick={() => setTheme("light")}>Light</DropdownMenuItem>
        <DropdownMenuItem onClick={() => setTheme("dark")}>Dark</DropdownMenuItem>
        <DropdownMenuItem onClick={() => setTheme("system")}>System</DropdownMenuItem>
      </DropdownMenuContent>
    </DropdownMenu>
  )
}
```

## Forms (react-hook-form + zod)

### Full Pattern

```tsx
"use client"
import { useForm } from "react-hook-form"
import { zodResolver } from "@hookform/resolvers/zod"
import { z } from "zod"
import { Form, FormControl, FormDescription, FormField, FormItem, FormLabel, FormMessage } from "@/components/ui/form"
import { Input } from "@/components/ui/input"
import { Button } from "@/components/ui/button"

const schema = z.object({
  email: z.string().email(),
  name: z.string().min(1).max(100),
})

type FormValues = z.infer<typeof schema>

export function ExampleForm() {
  const form = useForm<FormValues>({
    resolver: zodResolver(schema),
    defaultValues: { email: "", name: "" },
  })

  function onSubmit(values: FormValues) {
    console.log(values)
  }

  return (
    <Form {...form}>
      <form onSubmit={form.handleSubmit(onSubmit)} className="space-y-4">
        <FormField
          control={form.control}
          name="email"
          render={({ field }) => (
            <FormItem>
              <FormLabel>Email</FormLabel>
              <FormControl><Input type="email" {...field} /></FormControl>
              <FormMessage />
            </FormItem>
          )}
        />
        <Button type="submit" disabled={form.formState.isSubmitting}>
          {form.formState.isSubmitting ? "Saving..." : "Save"}
        </Button>
      </form>
    </Form>
  )
}
```

### Server Action Form

```tsx
// app/actions/example.ts
"use server"
import { z } from "zod"
import { revalidatePath } from "next/cache"

const schema = z.object({ name: z.string().min(1) })

export async function submitForm(data: z.infer<typeof schema>) {
  const result = schema.safeParse(data)
  if (!result.success) return { error: result.error.flatten().fieldErrors }
  // db call
  revalidatePath("/")
  return { success: true }
}
```

### Token Update Workflow

When Figma tokens change:
1. Re-export `variables-export.json` (lazyyysync plugin)
2. Convert RGB float → HSL: `L = (r+b)/2`, `S = range/(1-|2L-1|)`
3. Update CSS variables in `globals.css`
4. Verify: `--card` dark ≠ `--background` dark (differ by 1 shade)

Full variable reference: see `TOKENS.md` (1,806 variables, 17 collections)

## Accessibility

- Icon-only buttons must have `<span className="sr-only">label</span>`
- Every `<Input>` must have a linked `<Label htmlFor="id">`
- `<Dialog>` and `<Sheet>` must always include `DialogTitle` / `SheetTitle`
- Use `asChild` when `<Button>` wraps `<Link>`
- Always add `suppressHydrationWarning` to `<html>` when using dark mode
- Semantic tokens (`bg-background`, etc.) switch automatically — never add `dark:` prefix to them
